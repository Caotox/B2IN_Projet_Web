server {
    listen 80;
    server_name localhost;  # Remplacer par ton domaine si tu en as un

    root /home/debian/Documents/B2IN_Projet_Web/;  # Remplacer par le chemin de ton site
    index index.html;

    location / {
        try_files $uri $uri/ =404;
    }

    location /api/ {
        proxy_pass http://localhost:3000/;  # Redirige vers ton serveur Node.js
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }
}
